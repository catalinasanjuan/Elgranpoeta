def editar_bodega(nombre, data):
    try:
        bodega = data_access.obtener_bodega_por_nombre(nombre)
        if bodega:
            data_access.editar_bodega(nombre, data['nombre'], data['capacidad'], data['calle'], data['numero'], data['ciudad'], data['region'])
            return True
        return False
    except Exception as e:
        print(f"Error al modificar bodega: {e}")
        return False

#data_access
    def modificar_bodega(self, nombre, nuevo_nombre, capacidad, calle, numero, ciudad, region):
        query = """
        UPDATE direccion d
        JOIN bodega b ON b.Direccion_FK = d.ID_Direccion
        SET d.Calle = %s, d.Numero = %s, d.Ciudad = %s, d.Region = %s, b.Nombre_Bodega = %s, b.Capacidad = %s
        WHERE b.Nombre_Bodega = %s
        """
        self.cursor.execute(query, (calle, numero, ciudad, region, nuevo_nombre, capacidad, nombre))
        self.db.commit()
